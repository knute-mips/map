594 void push_back(const ReportDescriptor & rd) {
595 //Internal book-keeping is only for Python shell workflows.
596 //We also do not perform any book-keeping for descriptors
597 //whose dest_file is "1" (print to stdout). That is the
598 //only use case where duplicate dest_file's is actually
599 //okay. We will add the descriptor to the collection, but
600 //any attempts to access this descriptor later (i.e. to
601 //change its parameters or to disable it from Python) will
602 //throw an exception that the descriptor "1" cannot be found.
603 if (rd.dest_file != "1") {
604 const std::string desc_name = getDescriptorName_(rd);
605 auto iter = indices_by_descriptor_name_.find(desc_name);
606 if (iter != indices_by_descriptor_name_.end()) {
607 //Descriptor found by this name. Throw if it is already
608 //enabled. Users can only replace disabled descriptors.
609 if (rep_descs_.at(iter->second).isEnabled()) {
610 throw SpartaException("Report descriptor named '") <<
611 desc_name << "' already exists in this configuration";
612 }
613 }
614 indices_by_descriptor_name_[desc_name] = rep_descs_.size();
615 }
616
617 //This incoming 'rd' variable could be a reference to a
618 //previously disabled descriptor:
619 //
620 // >>> rd = report_config.descriptors.foo_csv
621 // >>> report_config.removeReport('foo_csv')
622 // >>> report_config.addReport(rd)
623 //
624 //This would still count as an enabled descriptor.
625 rep_descs_.push_back(rd);
626 rep_descs_.back().enabled_ = true;
627 }

661 bool contains(const std::string & desc_name) const {
662 auto iter = indices_by_descriptor_name_.find(desc_name);
663 if (iter == indices_by_descriptor_name_.end()) {
664 //The name may have been given to us with a period, not
665 //an underscore, separating the file stem and the extension
666 const auto replaced_desc_name = replaceDotsWithUnderscores_(desc_name);
667 iter = indices_by_descriptor_name_.find(replaced_desc_name);
668 }
669
670 //If this name is not even found in our map, we do not
671 //contain this descriptor
672 if (iter == indices_by_descriptor_name_.end()) {
673 return false;
674 }
675
676 //But even if the descriptor is physcially here, we have
677 //to also ask it if is is still enabled
678 return rep_descs_.at(iter->second).isEnabled();
679 }

736 size_t getDescriptorIndexByName_(const std::string & desc_name) const {
737 auto iter = indices_by_descriptor_name_.find(desc_name);
738 if (iter == indices_by_descriptor_name_.end()) {
739 //The name may have been given to us with a period, not
740 //an underscore, separating the file stem and the extension
741 const auto replaced_desc_name = replaceDotsWithUnderscores_(desc_name);
742 iter = indices_by_descriptor_name_.find(replaced_desc_name);
743
744 //If it still is not found, error out
745 if (iter == indices_by_descriptor_name_.end()) {
746 throw SpartaException(
747 "No descriptor named '") << desc_name << "' exists";
748 }
749 }
750
751 //The descriptor exists in this collection, but we still need
752 //to throw if it has been disabled, since we are not supposed
753 //to be using those descriptors for any reason anymore.
754 if (!rep_descs_.at(iter->second).isEnabled()) {
755 throw SpartaException("The descriptor named '") << desc_name
756 << "' has already been disabled";
757 }
758 return iter->second;
759 }
